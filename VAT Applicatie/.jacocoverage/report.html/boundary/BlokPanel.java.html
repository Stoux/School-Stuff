    @Override
    public boolean voegToe() {
        fout = ""; foutGevonden = false;
        //Check geldigheid van waardes
        if (!vormControle.isGeldigeDouble(lengteField.getText())) addFout("lengte");
        if (!vormControle.isGeldigeDouble(breedteField.getText())) addFout("breedte");
        if (!vormControle.isGeldigeDouble(hoogteField.getText())) addFout("hoogte");
        if (foutGevonden) {
            laatFoutmeldingZien();
            return false;
        }
        //Maak blok aan
        vormControle.maakBlok(vormControle.parseDouble(lengteField.getText()), vormControle.parseDouble(breedteField.getText()), vormControle.parseDouble(hoogteField.getText()));
        return true;
    }
